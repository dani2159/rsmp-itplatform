function GuideSideMenu({ currentFile }) {
  return (
    <aside className="guide-side-menu" aria-label="Menu guide design system">
      <div className="guide-side-menu-card">
        <p className="guide-side-menu-kicker">Design System</p>
        <strong className="guide-side-menu-title">SIMRS Guide</strong>
        <nav className="guide-side-menu-list">
          {guideRoutes.map((item) => (
            <a
              key={item.file}
              href={item.file.split('/').pop()}
              className={`guide-side-menu-link ${currentFile === item.file ? 'active' : ''}`}
            >
              {item.label}
            </a>
          ))}
        </nav>
      </div>
    </aside>
  )
}
